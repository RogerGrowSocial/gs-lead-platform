/**
 * Verrijk menu-items met flags - ROBUUSTE VERSIE
 */
add_filter('wp_nav_menu_objects', function ($items, $args) {
    if (empty($items) || !is_array($items)) return $items;

    // Build parent lookup
    $by_id = [];
    foreach ($items as $it) {
        $by_id[$it->ID] = $it;
    }

    // Find "Aanbiedingen" parent ID(s) - check multiple ways
    $aanbiedingen_ids = [];
    foreach ($items as $it) {
        $title = trim(wp_strip_all_tags($it->title));
        $title_lower = strtolower($title);
        
        // Check exact match or variations
        if (
            strcasecmp($title, 'Aanbiedingen') === 0 ||
            $title_lower === 'aanbiedingen' ||
            strpos($title_lower, 'aanbieding') !== false
        ) {
            $aanbiedingen_ids[] = $it->ID;
        }
    }

    foreach ($items as $it) {
        // Check if product category with switcher ON (uitgelichte_aanbiedingen)
        $is_featured = false;
        if ($it->object === 'product_cat' && !empty($it->object_id)) {
            $term_id = (int) $it->object_id;
            $flag = get_term_meta($term_id, 'uitgelichte_aanbiedingen', true);
            
            // Debug logging
            if (defined('WP_DEBUG') && WP_DEBUG) {
                $all_meta = get_term_meta($term_id);
                error_log("wp_nav_menu_objects filter: term_id={$term_id}, uitgelichte_aanbiedingen=" . var_export($flag, true) . ", all_meta_keys=" . implode(', ', array_keys($all_meta)));
            }
            
            // Multiple format checks
            $is_featured = (
                $flag === '1' || $flag === 1 || $flag === true ||
                $flag === 'true' || $flag === 'on' || $flag === 'yes' ||
                (is_numeric($flag) && (int) $flag === 1) ||
                $flag === 'checked' || $flag === 'enabled'
            );
        }
        $it->lammy_is_featured_cat = $is_featured;

        // Check if under "Aanbiedingen" - multiple methods
        $under = false;
        
        // Method 1: Check if parent ID is in aanbiedingen_ids
        if (!empty($it->menu_item_parent) && in_array((int) $it->menu_item_parent, $aanbiedingen_ids)) {
            $under = true;
        }
        
        // Method 2: Walk up parent chain
        if (!$under) {
            $cursor = $it;
            $max_depth = 10; // Prevent infinite loops
            $depth = 0;
            while ($cursor && !empty($cursor->menu_item_parent) && $depth < $max_depth) {
                $parent = $by_id[$cursor->menu_item_parent] ?? null;
                if (!$parent) break;
                
                $parent_title = trim(wp_strip_all_tags($parent->title));
                $parent_title_lower = strtolower($parent_title);
                
                if (
                    strcasecmp($parent_title, 'Aanbiedingen') === 0 ||
                    $parent_title_lower === 'aanbiedingen' ||
                    strpos($parent_title_lower, 'aanbieding') !== false ||
                    in_array($parent->ID, $aanbiedingen_ids)
                ) {
                    $under = true;
                    break;
                }
                
                $cursor = $parent;
                $depth++;
            }
        }
        
        // Method 3: Check menu location/args if available
        if (!$under && isset($args->theme_location)) {
            $location = $args->theme_location;
            // If it's a mega menu location, check if we're in the right section
            // This is a fallback - adjust based on your theme
        }

        $it->lammy_under_aanbiedingen = $under;
    }

    return $items;
}, 20, 2); // Higher priority

/**
 * Filter shortcode output voor lammy_aanbiedingen_megamenu
 * Voegt badges toe aan categorieÃ«n met uitgelichte_aanbiedingen = true
 * Gebruikt regex voor betrouwbaarheid
 */
add_filter('do_shortcode_tag', function ($output, $tag, $attr) {
    // Alleen voor de aanbiedingen megamenu shortcode
    if ($tag !== 'lammy_aanbiedingen_megamenu') {
        return $output;
    }
    
    // Parse de HTML output met regex
    if (empty($output) || strpos($output, 'mega-menu-links') === false) {
        return $output;
    }
    
    // Vind alle links die naar product categorieÃ«n gaan
    // Pattern: <a href="...product-categorie/slug...">tekst</a>
    $pattern = '/<a\s+href=["\']([^"\']*product-categorie[\/]([^"\'\/]+)[^"\']*)["\'][^>]*>([^<]*)<\/a>/i';
    
    $output = preg_replace_callback($pattern, function ($matches) {
        $full_match = $matches[0];
        $href = $matches[1];
        $slug = $matches[2];
        $link_text = $matches[3];
        
        // Check of er al een badge is (van codelammy.php of vorige run)
        if (strpos($full_match, 'ðŸ¤©') !== false || strpos($full_match, 'lammy-badge-uitgelicht') !== false) {
            return $full_match; // Al heeft badge
        }
        
        // Get term by slug
        $term = get_term_by('slug', $slug, 'product_cat');
        
        if ($term && !is_wp_error($term)) {
            // Check of deze categorie uitgelicht is
            $flag = get_term_meta($term->term_id, 'uitgelichte_aanbiedingen', true);
            
            // Debug logging
            if (defined('WP_DEBUG') && WP_DEBUG) {
                error_log("Shortcode filter: term={$term->name} (slug:{$slug}, ID:{$term->term_id}), flag=" . var_export($flag, true));
            }
            
            $is_featured = (
                $flag === '1' || $flag === 1 || $flag === true ||
                $flag === 'true' || $flag === 'on' || $flag === 'yes' ||
                (is_numeric($flag) && (int) $flag === 1) ||
                $flag === 'checked' || $flag === 'enabled'
            );
            
            if ($is_featured) {
                // Voeg badge toe voor de link tekst
                $pill = '<span class="lammy-mega-pill lammy-mega-pill--new" aria-label="Uitgelichte categorie">ðŸ¤© Nieuw</span> ';
                
                // Replace de link tekst met badge + tekst
                $new_link = str_replace('>' . $link_text . '<', '>' . $pill . $link_text . '<', $full_match);
                return $new_link;
            }
        }
        
        return $full_match;
    }, $output);
    
    return $output;
}, 10, 3);

/**
 * Cache flush functie - roep aan om cache te legen na wijzigingen
 * Gebruik: do_action('lammy_flush_megamenu_cache');
 */
add_action('lammy_flush_megamenu_cache', function() {
    // Flush transients
    delete_transient('lammy_aanbiedingen_megamenu_v2');
    delete_transient('lammy_collectie_megamenu_v3');
    
    // Flush alle lammy_megacol transients
    global $wpdb;
    if (isset($wpdb->options)) {
        $wpdb->query("DELETE FROM {$wpdb->options} WHERE option_name LIKE '_transient_lammy_megacol_%'");
        $wpdb->query("DELETE FROM {$wpdb->options} WHERE option_name LIKE '_transient_timeout_lammy_megacol_%'");
        $wpdb->query("DELETE FROM {$wpdb->options} WHERE option_name LIKE '_transient_lammy_aanbiedingen_megamenu_%'");
        $wpdb->query("DELETE FROM {$wpdb->options} WHERE option_name LIKE '_transient_timeout_lammy_aanbiedingen_megamenu_%'");
    }
    
    // Bump cache version als functie bestaat
    if (function_exists('lammy_megamenu_bump_cache_version')) {
        lammy_megamenu_bump_cache_version();
    }
    
    // Flush object cache
    if (function_exists('wp_cache_flush')) {
        wp_cache_flush();
    }
    
    if (defined('WP_DEBUG') && WP_DEBUG) {
        error_log('Lammy megamenu cache manually flushed');
    }
});

/**
 * Auto-flush cache wanneer term meta wordt gewijzigd
 */
add_action('updated_term_meta', function($meta_id, $object_id, $meta_key) {
    if ($meta_key === 'uitgelichte_aanbiedingen') {
        do_action('lammy_flush_megamenu_cache');
        if (defined('WP_DEBUG') && WP_DEBUG) {
            error_log("Cache auto-flushed: updated_term_meta voor term_id={$object_id}, meta_key={$meta_key}");
        }
    }
}, 10, 3);

add_action('added_term_meta', function($meta_id, $object_id, $meta_key) {
    if ($meta_key === 'uitgelichte_aanbiedingen') {
        do_action('lammy_flush_megamenu_cache');
        if (defined('WP_DEBUG') && WP_DEBUG) {
            error_log("Cache auto-flushed: added_term_meta voor term_id={$object_id}, meta_key={$meta_key}");
        }
    }
}, 10, 3);
